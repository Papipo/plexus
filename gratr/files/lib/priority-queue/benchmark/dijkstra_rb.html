53: def dijkstra(start_node, queue_klass)
54:   # Priority Queue with unfinished nodes
55:   active = queue_klass.new
56:   # Distances for all nodes
57:   distances = Hash.new { 1.0 / 0.0 }
58:   # Parent pointers describing shortest paths for all nodes
59:   parents = Hash.new
60: 
61:   # Initialize with start node
62:   active[start_node] = 0
63:   until active.empty?
64:     u, distance = active.delete_min
65:     distances[u] = distance
66:     d = distance + 1
67:     u.neighbours.each do | v |
68:       next unless d < distances[v] # we can't relax this one
69:       active[v] = distances[v] = d
70:       parents[v] = u
71:     end    
72:   end
73: end
